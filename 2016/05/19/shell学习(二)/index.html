shell学习(二)

shell编程中字符串的操作

  • 连接字符串

    str=hello
str1=world
连接起来：str=str$str1、str = $str' '"aa123"、str = $str:

  • 查看字符串长度：

    • expr length "hello"
    • expr length "$str"
    • charcount=\expr length “$str”``
    • echo $charcount
  • 查看特定字符：
    • expr index "$str" 'h' #从1开始，不是0，只返回第一个。如果是查找多个字母，只返回这些字母的第一个查找到的位置
  • 字符串截断：
    • expr substr "$str" 6 10 #获取指定位置指定长度的字符串
  • 正则匹配
    • expr $str : REGEXP
    • expr match $str REGEXP
    • expr $str : '.*' #返回长度
    • expr $str : '(.*\[0-9]\{5\}\)'
    • expr $str : '\([a-zA-X: ]*\).*'

shell编程中的数学运算

  • 数值运算：+、-、*、/、%、()
  • expr expression
  • result=$[expression]
  • 例子：
    num1=7
num2=13
num3=7
expr $num1 \< $num2
expr $num1 \< 9
result =$[$sum1<9] #不用转义
echo $result
expr $num1 = $num3 #等号两边必须有空隔
result = $[$num1 == 7] #在中括号中必须使用双等号
expr $num1 \* $num2
result=$[$num1*$num2]
result=$[$num1/$num2]
  • 浮点运算：通过内建计算器：bc,bc能够识别：
    • 数字（整形和浮点型）
    • 变量
    • 注释（以#开始的行，或者c语言中的注释形式/ /）
    • 表达式
    • 编程语句（如条件判断：if-then）
    • 函数
  • bc：可以输出版本、版权等，然后进入bc，可以进行浮点计算。首先设置小数位数：scale=4,quit退出
  • bc中也支持变量：
    num=12
num2=3;num3=4
num/num2

  • 在脚本中使用bc：

    var = `echo "options;expression" | bc`
var = `echo "scale=4;3.44/5" | bc` //管道方式
echo $var

var1=10.46
var2=43.34
var3=32.3
VAR4=54.2
result=`bc << EOF  //内联方式
>scale=4
>a1 = $var1+$var2
>b1 = $var3+$var4
>a1+b1
>EOF
>`

shell编程中的循环-for

  • 语法结构

    for var in list
do
    commands
done

  • 一个例子：

    #! /bin/bash
#直接列表
for val in Jan Feb Mar Apr May
do
    echo "Month name is $val"
done

#从变量读取
list="Jan Feb Mar Apr May"
for val in $list
do
    echo "Month name is $val"
done

#从命令读取
for val in `cat ltestfile`  #反引号，执行并返回。获取文件内容并返回
do
    echo "Month name is $val in file"
done

#! /bin/bash
#从目录读取,获取所有的文件,/home/c*,/home/*c
for val in /home/*
do
    echo "$val"
done

for val in `ls /home/*`
do
    echo "$val"
done

  • 语法结构

    for((i=1;i<=10;i++))
do
    commands
done

  • 一个例子：

    #! /bin/bash
#直接列表
for ((i=1,j=10;i<=10;i++,j--))
do
    echo "number is $i $j"
done

shell编程中的循环-while、until

  • 语法结构
    while test command #test执行成功，返回0，执行不成功，返回1。判断成立时，执行语句。
    do
    commands
    done

until test command #test执行成功，返回0，执行不成功，返回1。判断成立时，结束执行
do
commands
done

  • 一个例子

    #! /bin/bash
var =1
while [$var -lt 10]
do
    echo "$var"
    var = $[$var +1]
done

var = 1
until [$var -eq 10]
do
    echo "$var"
    var = $[$var +1]
done

shell编程中的循环嵌套

  • 例子：

    #! /bin/bash
for((i=1;i<=3;i++))
do
    echo "Outside loop:$i"
    for((j=1;j<=3;j++))
    do
        echo "inside loop:$j"
    done
done

i=1
while [$i -le 3]
do
    echo "Outside loop:$i"
    for((j=1;j<=3;j++))
    do
        echo "inside loop:$j"
    done
    i=$[$i+1] #赋值语句不能有空隔
done

  • break：跳出循环

    #! /bin/bash
for((i=1;i<=10;i++))
do
    if [$i -eq 5]
    then 
        break;
    fi
    echo "$i"
done

  • continue：继续循环

    #! /bin/bash
for((i=1;i<=10;i++))
do
    if [$i -eq 5]
    then 
        continue;
    fi
    echo "$i"
done
